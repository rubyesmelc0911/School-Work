#include <iostream>
using namespace std;

void quicksort(int arr[], int low, int high) {

    if (low < high) {
        int i = low, j = high;
        int pivote = arr[(low + high) / 2];
        int temp;

        while (i <= j) {
            while (arr[i] < pivote) i++;
            while (arr[j] > pivote) j--;

            if (i <= j) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
                i++;
                j--;
            }
        }

        // Llamadas recursivas correctas
        quicksort(arr, low, j);
        quicksort(arr, i, high);
    }
}

int main() {

    int arr[] = {};
    int n = sizeof(arr) / sizeof(arr[0]);

    cout << "Arreglo original: ";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";

    quicksort(arr, 0, n - 1);

    cout << "\n\nArreglo ordenado (QuickSort): ";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";

    return 0;
}
